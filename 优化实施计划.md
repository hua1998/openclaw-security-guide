```python
"""security_detector.py 单元测试"""
import unittest
from pathlib import Path
import json
import sys

sys.path.insert(0, str(Path(__file__).parent.parent / 'tools'))

from security_detector import load_config, check_auth, calculate_score

class TestSecurityDetector(unittest.TestCase):
    def setUp(self):
        self.test_config = {
            "gateway": {
                "auth": {
                    "mode": "token",
                    "token": "a" * 32
                }
            },
            "session": {
                "dmScope": "per-channel-peer"
            },
            "tools": {
                "profile": "minimal",
                "deny": ["group:automation", "group:runtime"]
            },
            "sandbox": {
                "enabled": True,
                "image": "docker.io/openclaw/sandbox:latest"
            },
            "audit": {
                "enabled": True
            }
        }
    
    def test_check_auth_with_valid_token(self):
        """测试有效Token认证"""
        issues = check_auth(self.test_config)
        auth_issues = [i for i in issues if i['category'] == 'authentication']
        self.assertEqual(len(auth_issues), 0)
    
    def test_check_auth_without_token(self):
        """测试缺少Token"""
        self.test_config['gateway']['auth']['mode'] = 'none'
        issues = check_auth(self.test_config)
        self.assertTrue(any(i['risk'] == 'critical' for i in issues))
    
    def test_check_auth_weak_token(self):
        """测试弱Token"""
        self.test_config['gateway']['auth']['token'] = 'short'
        issues = check_auth(self.test_config)
        self.assertTrue(any('length' in i.get('check', '') for i in issues))

if __name__ == '__main__':
    unittest.main()
```

#### 7.2 Dify 适配器设计

**文件**: `tools/adapters/dify_adapter.py`
```python
"""Dify 平台配置适配器"""
import os
from pathlib import Path
import yaml

class DifyAdapter:
    """将Dify配置转换为OpenClaw安全检测格式"""
    
    def __init__(self, config_path: str):
        self.config_path = Path(config_path)
        self.raw_config = self._load_config()
    
    def _load_config(self) -> dict:
        """加载Dify配置"""
        # 支持 .env 和 docker-compose.yaml
        if self.config_path.suffix == '.env':
            return self._parse_env()
        elif self.config_path.name == 'docker-compose.yaml':
            return self._parse_compose()
        else:
            raise ValueError(f"Unsupported config format: {self.config_path}")
    
    def _parse_env(self) -> dict:
        """解析.env文件"""
        config = {}
        with open(self.config_path, 'r') as f:
            for line in f:
                if '=' in line and not line.startswith('#'):
                    key, value = line.strip().split('=', 1)
                    config[key] = value
        return self._transform_env_config(config)
    
    def _transform_env_config(self, env_config: dict) -> dict:
        """转换为OpenClaw格式"""
        return {
            "gateway": {
                "auth": {
                    "mode": "api_key" if env_config.get('APP_API_KEY') else "none",
                    "token": env_config.get('APP_API_KEY', '')
                }
            },
            "sandbox": {
                "enabled": env_config.get('CODE_EXECUTION_ENABLED', 'false').lower() == 'true',
                "runtime": "docker"
            },
            "audit": {
                "enabled": True,
                "destination": [{
                    "type": "file",
                    "path": env_config.get('LOG_FILE', '/var/log/dify/app.log')
                }]
            }
        }
    
    def to_openclaw_format(self) -> dict:
        """导出为OpenClaw标准格式"""
        return self.raw_config
```

#### 7.3 多平台检测入口

**文件**: `tools/multi_platform_scanner.py`
```python
#!/usr/bin/env python3
"""多平台安全扫描器入口"""
import argparse
import sys
from pathlib import Path

# 导入适配器
from adapters.dify_adapter import DifyAdapter
from adapters.autogpt_adapter import AutoGPTAdapter
from adapters.fastgpt_adapter import FastGPTAdapter

ADAPTERS = {
    'openclaw': None,  # 原生支持
    'dify': DifyAdapter,
    'autogpt': AutoGPTAdapter,
    'fastgpt': FastGPTAdapter,
}

def detect_platform(config_path: str) -> str:
    """自动检测平台类型"""
    path = Path(config_path)
    
    if 'dify' in path.name.lower() or path.name == '.env':
        return 'dify'
    elif path.name == 'auto-gpt.json' or '.autogpt' in str(path):
        return 'autogpt'
    elif path.name == 'config.yaml' and 'fastgpt' in str(path).lower():
        return 'fastgpt'
    elif path.suffix == '.json':
        return 'openclaw'
    
    return 'unknown'

def main():
    parser = argparse.ArgumentParser(description='多平台AI Agent安全扫描器')
    parser.add_argument('--config', required=True, help='配置文件路径')
    parser.add_argument('--platform', choices=list(ADAPTERS.keys()), 
                       help='指定平台类型（自动检测）')
    parser.add_argument('--json', action='store_true', help='JSON格式输出')
    args = parser.parse_args()
    
    # 自动检测平台
    platform = args.platform or detect_platform(args.config)
    if platform == 'unknown':
        print("错误: 无法自动检测平台类型，请使用 --platform 指定")
        sys.exit(1)
    
    print(f"检测到平台: {platform}")
    
    # 加载适配器
    adapter_class = ADAPTERS.get(platform)
    if adapter_class:
        adapter = adapter_class(args.config)
        openclaw_config = adapter.to_openclaw_format()
        
        # 保存临时配置文件
        temp_config = Path('/tmp/openclaw_temp_config.json')
        import json
        with open(temp_config, 'w') as f:
            json.dump(openclaw_config, f, indent=2)
        
        # 调用原生检测器
        import subprocess
        cmd = ['python', 'tools/security_detector.py', '--config', str(temp_config)]
        if args.json:
            cmd.append('--json')
        
        result = subprocess.run(cmd, capture_output=True, text=True)
        print(result.stdout)
        if result.returncode != 0:
            print(result.stderr)
            sys.exit(result.returncode)
    else:
        # 原生OpenClaw直接检测
        import subprocess
        cmd = ['python', 'tools/security_detector.py', '--config', args.config]
        if args.json:
            cmd.append('--json')
        subprocess.run(cmd)

if __name__ == '__main__':
    main()
```
